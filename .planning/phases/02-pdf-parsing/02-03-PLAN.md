Create `apps/api/src/middleware/upload.middleware.ts` — a composed middleware array that handles multer file reception, size enforcement, and two-layer PDF type validation (MIME header + magic bytes).

Purpose: File validation must happen before any parsing attempt. Multer enforces the 10MB limit natively (throws `MulterError`), but MIME-type validation alone is spoofable — the magic bytes check (`%PDF`) provides a second layer. Both layers produce specific, differentiated errors so the frontend can show targeted messages (INPUT-02, INPUT-03).

Output: `uploadMiddleware` exported as `RequestHandler[]` — ready to spread into a route definition.

    ```typescript
    import multer, { MulterError } from 'multer';
    import type { Request, Response, NextFunction, RequestHandler } from 'express';
    import { PdfNotPdfError, PdfTooLargeError } from './error.middleware.js';

    const PDF_MAGIC = Buffer.from([0x25, 0x50, 0x44, 0x46]); // %PDF

    const storage = multer.memoryStorage();
    const multerUpload = multer({
      storage,
      limits: { fileSize: 10 * 1024 * 1024 },
      fileFilter: (_req, file, cb) => {
        if (file.mimetype === 'application/pdf') {
          cb(null, true);
        } else {
          cb(null, false); // silent reject — validateMagicBytes throws specific error
        }
      },
    }).single('resume');

    function wrapMulter(req: Request, res: Response, next: NextFunction): void {
      multerUpload(req, res, (err) => {
        if (err instanceof MulterError) {
          if (err.code === 'LIMIT_FILE_SIZE') {
            next(new PdfTooLargeError('File exceeds the 10MB size limit.'));
          } else {
            next(err);
          }
        } else if (err) {
          next(err);
        } else {
          next();
        }
      });
    }

    function validateMagicBytes(req: Request, _res: Response, next: NextFunction): void {
      if (!req.file) {
        next(new PdfNotPdfError('No file uploaded or file type is not PDF.'));
        return;
      }
      const head = req.file.buffer.slice(0, 4);
      if (!head.equals(PDF_MAGIC)) {
        next(new PdfNotPdfError('File is not a valid PDF (magic bytes mismatch).'));
        return;
      }
      next();
    }

    export const uploadMiddleware: RequestHandler[] = [wrapMulter, validateMagicBytes];
    ```
